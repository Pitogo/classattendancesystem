// ClassRFID — Classroom Attendance System
// Hardware: Arduino UNO + MFRC522 + 16x2 LCD + Buzzer

#include <SPI.h>
#include <MFRC522.h>
#include <LiquidCrystal_I2C.h>
#include <SD.h>

#define RST_PIN    9
#define SS_PIN     10
#define BUZZER_PIN 8
#define GREEN_LED  7
#define RED_LED    6
#define SD_CS_PIN  4

MFRC522           rfid(SS_PIN, RST_PIN);
LiquidCrystal_I2C lcd(0x27, 16, 2);

struct Student { String uid; String name; String studentId; };

const int NUM_STUDENTS = 9;
Student students[NUM_STUDENTS] = {
  { "A1B2C3D4", "Gabriel del Mundo",        "STU-001" },
  { "E5F6G7H8", "Floyd Dillera",            "STU-002" },
  { "I9J0K1L2", "Maverick Ferrer",          "STU-003" },
  { "M3N4O5P6", "Austin Honra",             "STU-004" },
  { "Q7R8S9T0", "Christian Allen Laguerta",  "STU-005" },
  { "U1V2W3X4", "Jamir Ninonuevo",          "STU-006" },
  { "Y5Z6A7B8", "Gerald Opalla",            "STU-007" },
  { "C9D0E1F2", "Brixs James Pitogo",       "STU-008" },
  { "G3H4I5J6", "Princess Rivera",          "STU-009" }
};

String logFileName = "attend.csv";
bool sdAvailable = false;

void setup() {
  Serial.begin(9600);
  SPI.begin(); rfid.PCD_Init();
  lcd.init(); lcd.backlight();
  lcd.print("ClassRFID v1.0");
  lcd.setCursor(0,1); lcd.print("Scan your card..");
  pinMode(BUZZER_PIN,OUTPUT); pinMode(GREEN_LED,OUTPUT); pinMode(RED_LED,OUTPUT);
  if(SD.begin(SD_CS_PIN)){ sdAvailable=true; initCSV(); }
  beep(1,100);
}

void loop() {
  if(!rfid.PICC_IsNewCardPresent()||!rfid.PICC_ReadCardSerial()) return;
  String uid = getUID();
  int idx = findStudent(uid);
  if(idx>=0) handlePresent(idx); else handleUnknown(uid);
  rfid.PICC_HaltA(); rfid.PCD_StopCrypto1(); delay(1500);
}

void handlePresent(int idx) {
  String name=students[idx].name, sid=students[idx].studentId;
  lcd.clear(); lcd.print(name); lcd.setCursor(0,1); lcd.print("Welcome! ✓");
  digitalWrite(GREEN_LED,HIGH); beep(2,80); delay(200); digitalWrite(GREEN_LED,LOW);
  logToSD(sid, name, getTimestamp(), "PRESENT");
}

void handleUnknown(String uid) {
  lcd.clear(); lcd.print("ACCESS DENIED"); lcd.setCursor(0,1); lcd.print("Unknown card!");
  digitalWrite(RED_LED,HIGH); beep(3,200); delay(200); digitalWrite(RED_LED,LOW);
}

String getUID() {
  String uid="";
  for(byte i=0;i<rfid.uid.size;i++){ if(rfid.uid.uidByte[i]<0x10) uid+="0"; uid+=String(rfid.uid.uidByte[i],HEX); }
  uid.toUpperCase(); return uid;
}

int findStudent(String uid) {
  for(int i=0;i<NUM_STUDENTS;i++) if(students[i].uid==uid) return i;
  return -1;
}

void beep(int t,int ms){ for(int i=0;i<t;i++){ tone(BUZZER_PIN,1000,ms); delay(ms+50); } }

String getTimestamp() {
  unsigned long ms=millis();
  return String((ms/3600000)%24)+":"+String((ms/60000)%60)+":"+String((ms/1000)%60);
}

void initCSV() {
  if(!SD.exists(logFileName)){ File f=SD.open(logFileName,FILE_WRITE); if(f){ f.println("StudentID,Name,Time,Status"); f.close(); } }
}

void logToSD(String sid,String name,String ts,String status) {
  if(!sdAvailable) return;
  File f=SD.open(logFileName,FILE_WRITE); if(f){ f.println(sid+","+name+","+ts+","+status); f.close(); }
}
